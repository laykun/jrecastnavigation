432 inline unsigned int dtIlog2(unsigned int v)
433 {
434  unsigned int r;
435  unsigned int shift;
436  r = (v > 0xffff) << 4; v >>= r;
437  shift = (v > 0xff) << 3; v >>= shift; r |= shift;
438  shift = (v > 0xf) << 2; v >>= shift; r |= shift;
439  shift = (v > 0x3) << 1; v >>= shift; r |= shift;
440  r |= (v >> 1);
441  return r;
442 }
